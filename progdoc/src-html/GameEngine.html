209    /**
210     * analyse ce que le joueur Ã©crit sur le clavier
211     * @param pCommand
212     */
213    public void interpretCommand(final String pCommand) 
214
215    {
216        this.aGui.println( "> " + pCommand );
217        Command vCommand = this.aParser.getCommand( pCommand );
218        String vCommandWord = vCommand.getCommandWord();
219
220        String vDirection= vCommand.getSecondWord();
221        Room vNextRoom = this.aPlayer.getCurrentRoom().getExit (vDirection);
222
223        if (vCommand.isUnknown() ) {
224            this.aGui.println("I don't know what you mean...");
225        } 
226        else if(vCommandWord.equals("help")){
227            printHelp();
228        }
229        else if(vCommandWord.equals("test")){
230            if ( vCommand.hasSecondWord() )
231            {
232                this.lecture(vCommand);
233            }
234        }
235        else if(vCommandWord.equals("go")){
236            goRoom(vCommand);
237        }
238        else if(vCommandWord.equals("look")){
239            look();
240        }
241        else if(vCommandWord.equals("quit")){
242            //boolean wantToQuit = quit(pCommand);
243            if ( vCommand.hasSecondWord() )
244                this.aGui.println( "Quit what?" );
245            else
246                this.endGame();
247        }
248
249        else if(vCommandWord.equals("back"))
250        {
251
252            if(vCommand.hasSecondWord()||this.aPlayer.getStack().empty()==true) 
253                this.aGui.println( "error" );
254            else{
255                aPlayer.goBack();
256                printLocationInfo();
257                if (  this.aPlayer.getCurrentRoom().getImageName() != null )
258                    this.aGui.showImage(  this.aPlayer.getCurrentRoom().getImageName() );
259            }
260        }
261        else if(vCommandWord.equals("take"))
262        {
263            take(vCommand);
264        }
265        else if(vCommandWord.equals("drop"))
266        {
267            drop(vCommand);
268        }
269        else if(vCommandWord.equals("eat")){
270            eat();
271        }
272        else
273            this.endGame();
274
275    }// effectue une mÃ©thode en fonction de la commande taper

277    /** permet de prendre un item
278     * @param pCommand
279    **/
280    public void take(final Command pCommand){
281        if (!pCommand.hasSecondWord())
282            this.aGui.println("take what???");
283        else{
284            String vItemString=pCommand.getSecondWord();
285            Room vPlayerRoom= this.aPlayer.getCurrentRoom();
286            Item vItem=(Item)vPlayerRoom.getItems().get(vItemString);
287
288            if (vItem==null)
289                this.aGui.println("There no items here");
290            else{
291                this.aPlayer.ptake(vItemString,vItem);
292                vPlayerRoom.removeItem(vItemString);
293
294            }
295        }
296    }

298     /** permet d'enlever un item
299     * @param pCommand
300    **/
301    public void drop(final Command pCommand){
302        if (!pCommand.hasSecondWord())
303            this.aGui.println("drop what???");
304        else{
305            String vItemString=pCommand.getSecondWord();
306            Room vPlayerRoom= this.aPlayer.getCurrentRoom();
307            Item vItem=(Item)this.aPlayer.getItemSac().get(vItemString);
308
309            if (this.aPlayer.getItemSac().isEmpty()==true)
310                this.aGui.println("There is nothing to drop");
311            else{
312                this.aPlayer.pdrop(vItemString);
313                vPlayerRoom.addItem(vItemString,vItem);
314
315            }
316        }
317    }
